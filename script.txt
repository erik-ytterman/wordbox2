import scala.io.Source
import scala.collection.mutable.MutableList

var words = Source.fromFile("/mnt/hgfs/Dump/Text/Wordlists/Swedish/ord.utf8").getLines.toList
var filtered = words.filter(_.length == 3)

val playstate = MutableList[String]()

var columnfilters = (word: String) => { for((chr, index) <- word.zipWithIndex) yield { playstate.map(_(index)).mkString + word(index).toString } }

var columnmatcher = (filters: List[String]) => {
  for(filter <- filters) yield {
    filtered.find(_.slice(0, playstate.length + 1) == filter) match { case Some(word) => true case None => false }
  }
}

var wordmatcher = (word: String) => { columnmatcher(columnfilters(word).toList).foldLeft(true)(_ && _) } : Boolean

var validwords = filtered.filter(wordmatcher(_))
