import scala.io.Source
import scala.collection.mutable.MutableList

var words = Source.fromFile("/mnt/hgfs/Dump/Text/Wordlists/Swedish/ord.utf8").getLines.toList
var filtered = words.filter(_.length == 5)

val playstate = MutableList[String]()

var columnfilters = (word: String) => {
  for((chr, index) <- word.zipWithIndex) yield { 
    var filterstring = playstate.map(_(index)).mkString + word(index).toString
    filtered.find(_.slice(0, playstate.length + 1) == filterstring) match { case Some(word) => true case None => false } 
  }
}

var wordfilter = (word: String) => { columnfilter(word).foldLeft(true)(_ && _) } : Boolean

var validwords = filtered.filter(wordselector(_))
